1. Initialize disc[u] = -1, low[u] = 0, time = 0
2. For each vertex u not visited:
   DFS(u, parent = -1)
DFS(u, parent):
   disc[u] = low[u] = ++time
   children = 0
   for v in adj[u]:
     if disc[v] == -1:
       children++
       mark edge (u,v) as tree-edge
       DFS(v, u)
       low[u] = min(low[u], low[v])
       if parent == -1 and children > 1:
         u is articulation point
       if parent != -1 and low[v] >= disc[u]:
         u is articulation point
       if low[v] > disc[u]:
         edge (u,v) is bridge
     else if v != parent:
       low[u] = min(low[u], disc[v])
       mark edge (u,v) as back-edge
